# Vidgen Service Commands

## Setup (run once)
sudo cp vidgen.service /etc/systemd/system/
sudo systemctl daemon-reload
sudo systemctl enable vidgen

## Start the service
sudo systemctl start vidgen

## Stop the service
sudo systemctl stop vidgen

## Restart the service
sudo systemctl restart vidgen

## Check status
sudo systemctl status vidgen

## View logs (live)
sudo journalctl -u vidgen -f

## View logs (last 100 lines)
sudo journalctl -u vidgen -n 100

## Disable service (prevent auto-start on boot)
sudo systemctl disable vidgen

## Kill the process manually (if needed)
sudo pkill -f "python3.*app.py"

## Switch between services
bash /root/switch_to_vidgen.sh      # Switch to vidgen
bash /root/switch_to_image_editor.sh # Switch to image-editor
